⚙️ Google Sheets Setup

Connect your Google Sheet to save all waitlist and payment data in real time. Takes about 5 minutes.

1

Create a Google Sheet

Go to sheets.google.com → create a new spreadsheet → name it LetterMatch Waitlist. Leave it empty — the script will add headers automatically.

2

Open Apps Script

In your Google Sheet, go to Extensions → Apps Script. Delete any existing code, then paste the script below:

function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); // Select sheet tab based on type const tabName = data.type === 'founding' ? 'Payments' : 'Waitlist'; let tab = sheet.getSheetByName(tabName); if (!tab) { tab = sheet.insertSheet(tabName); if (tabName === 'Waitlist') { tab.appendRow(['Timestamp','Email','Question','Answer']); } else { tab.appendRow(['Timestamp','Email','Name','Amount','Tier','OrderID']); } } if (tabName === 'Waitlist') { tab.appendRow([data.timestamp, data.email, data.question, data.answer]); } else { tab.appendRow([data.timestamp, data.email, data.name, data.amount, data.tier, data.orderId]); } return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { if (e.parameter.action !== 'stats') { return ContentService.createTextOutput('{}'); } const ss = SpreadsheetApp.getActiveSpreadsheet(); const wl = ss.getSheetByName('Waitlist'); const pm = ss.getSheetByName('Payments'); const waitlistCount = wl ? Math.max(0, wl.getLastRow() - 1) : 0; const foundingCount = pm ? Math.max(0, pm.getLastRow() - 1) : 0; let revenue = 0; let lastName = 'Ayala'; let lastTime = '2h ago'; if (pm && pm.getLastRow() > 1) { const amounts = pm.getRange(2, 4, pm.getLastRow()-1, 1).getValues(); amounts.forEach(r => revenue += parseFloat(r[0]) || 0); } if (wl && wl.getLastRow() > 1) { const lastRow = wl.getRange(wl.getLastRow(), 1, 1, 2).getValues()[0]; const email = lastRow[1] || ''; lastName = email.split('@')[0].split('.')[0]; lastName = lastName.charAt(0).toUpperCase() + lastName.slice(1); const ts = new Date(lastRow[0]); const diffMins = Math.floor((new Date() - ts) / 60000); lastTime = diffMins < 60 ? diffMins + 'm ago' : Math.floor(diffMins/60) + 'h ago'; } const result = { waitlist: waitlistCount, founding: foundingCount, revenue: revenue.toFixed(2), lastName, lastTime }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
3

Deploy as Web App

Click Deploy → New deployment. Choose type: Web app. Set "Execute as" → Me. Set "Who has access" → Anyone. Click Deploy → copy the URL.

4

Paste your URL here & in index.html

Paste your Apps Script deployment URL below to test it, and also replace YOUR_GOOGLE_APPS_SCRIPT_URL_HERE in index.html.

⚙️ Google Sheets Setup

Connect your Google Sheet to save all waitlist and payment data in real time. Takes about 5 minutes.

1

Create a Google Sheet

Go to sheets.google.com → create a new spreadsheet → name it LetterMatch Waitlist. Leave it empty — the script will add headers automatically.

2

Open Apps Script

In your Google Sheet, go to Extensions → Apps Script. Delete any existing code, then paste the script below:

function jsonResponse_(payload) { return ContentService.createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } function getOrCreateTab_(ss, name, headers) { var tab = ss.getSheetByName(name); if (!tab) { tab = ss.insertSheet(name); tab.appendRow(headers); } return tab; } function getWaitlistRows_(tab) { if (!tab || tab.getLastRow() <= 1) return []; return tab.getRange(2, 1, tab.getLastRow() - 1, 4).getValues().map(function(row) { return { timestamp: row[0], email: row[1], question: row[2], answer: row[3], type: 'waitlist' }; }); } function getPaymentRows_(tab) { if (!tab || tab.getLastRow() <= 1) return []; return tab.getRange(2, 1, tab.getLastRow() - 1, 6).getValues().map(function(row) { return { timestamp: row[0], email: row[1], name: row[2], amount: String(row[3] || ''), tier: row[4], orderId: row[5], type: 'founding' }; }); } function sortNewestFirst_(rows) { return rows.sort(function(a, b) { return new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime(); }); } function buildStats_(waitlistRows, paymentRows) { var revenue = paymentRows.reduce(function(sum, row) { return sum + (parseFloat(row.amount) || 0); }, 0); var lastName = '—'; var lastTime = '—'; if (waitlistRows.length) { var latest = sortNewestFirst_(waitlistRows.slice())[0]; var email = String(latest.email || ''); var localPart = email.split('@')[0].split('.')[0]; lastName = localPart ? localPart.charAt(0).toUpperCase() + localPart.slice(1) : '—'; var diffMins = Math.floor((new Date() - new Date(latest.timestamp)) / 60000); lastTime = diffMins < 60 ? diffMins + 'm ago' : Math.floor(diffMins / 60) + 'h ago'; } return { waitlist: waitlistRows.length, founding: paymentRows.length, revenue: revenue.toFixed(2), lastName: lastName, lastTime: lastTime }; } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents || '{}'); var isFounding = data.type === 'founding'; var tab = getOrCreateTab_( ss, isFounding ? 'Payments' : 'Waitlist', isFounding ? ['Timestamp','Email','Name','Amount','Tier','OrderID'] : ['Timestamp','Email','Question','Answer'] ); if (isFounding) { tab.appendRow([data.timestamp, data.email, data.name, data.amount, data.tier, data.orderId]); } else { tab.appendRow([data.timestamp, data.email, data.question, data.answer]); } return jsonResponse_({ status:'ok' }); } function doGet(e) { var action = (e && e.parameter && e.parameter.action) || 'stats'; var ss = SpreadsheetApp.getActiveSpreadsheet(); var waitlistRows = sortNewestFirst_(getWaitlistRows_(ss.getSheetByName('Waitlist'))); var paymentRows = sortNewestFirst_(getPaymentRows_(ss.getSheetByName('Payments'))); var stats = buildStats_(waitlistRows, paymentRows); if (action === 'dashboard') { return jsonResponse_({ status: 'ok', stats: stats, entries: sortNewestFirst_(waitlistRows.concat(paymentRows)), payments: paymentRows }); } if (action === 'stats') { return jsonResponse_(stats); } return jsonResponse_({ status:'ok' }); }
3

Deploy as Web App

Click Deploy → New deployment. Choose type: Web app. Set "Execute as" → Me. Set "Who has access" → Anyone. Click Deploy → copy the URL.

4

Paste your URL here & in index.html

Paste your Apps Script deployment URL below to test it, and also replace YOUR_GOOGLE_APPS_SCRIPT_URL_HERE in index.html.
